# ShopCopilot 店铺助手前端开发计划

## 项目概述

**项目名称**：shop-copilot-chat

**项目定位**：店铺智能助手聊天界面，为店长、导玩员、仓管、财务提供智能问答、数据查询、经营分析等服务。

**技术栈**：
- 框架：Vue3 + TypeScript
- 构建工具：Vite
- UI 库：Element Plus
- 状态管理：Pinia
- Markdown 渲染：marked + DOMPurify

## API 接口设计

### 1. 聊天接口（SSE 流式）

```typescript
// api/chat.ts
export function chatStream(
  message: string,
  sessionId: string,
  imageUrl?: string
): EventSource {
  const baseUrl = import.meta.env.VITE_API_BASE_URL || ''
  const url = `${baseUrl}/api/chat/stream`
  const params = new URLSearchParams({
    message,
    session_id: sessionId,
    ...(imageUrl && { image_url: imageUrl })
  })
  
  return new EventSource(`${url}?${params}`)
}
```

**SSE 数据格式**：
```typescript
interface SSEData {
  type: 'thinking' | 'processing' | 'tool_result' | 'answer' | 'done' | 'error'
  content: string
  step: string
  done: boolean
}
```

```typescript
export async function getSessions(): Promise<Session[]> {

  return response.data
}

// 创建会话
export async function createSession(title?: string): Promise<Session> {
  const response = await http.post('/api/chat/sessions', { title })
  return response.data
}

// 删除会话
export async function deleteSession(sessionId: string): Promise<void> {
  await http.delete(`/api/chat/sessions/${sessionId}`)
}

// 获取会话历史
export async function getSessionHistory(sessionId: string): Promise<Message[]> {
  const response = await http.get(`/api/chat/sessions/${sessionId}/messages`)
  return response.data
}
```

## 状态管理设计

### Pinia Store

```typescript
// stores/chat.ts
import { defineStore } from 'pinia'

interface ChatState {
  // 会话相关
  sessions: Session[]
  currentSessionId: string | null
  
  // 消息相关
  messages: Message[]
  isLoading: boolean
  
  // UI 状态
  isFullscreen: boolean
  showSessionList: boolean
  showQuickQuestions: boolean
  
  // 用户信息
  token: string
  shopId: number
  role: string
  userName: string
}

export const useChatStore = defineStore('chat', {
  state: (): ChatState => ({
    sessions: [],
    currentSessionId: null,
    messages: [],
    isLoading: false,
    isFullscreen: false,
    showSessionList: true,
    showQuickQuestions: true,
    token: '',
    shopId: 0,
    role: '',
    userName: '',
  }),
  
  getters: {
    currentSession: (state) => {
      return state.sessions.find(s => s.id === state.currentSessionId)
    },
    
    sortedSessions: (state) => {
      return [...state.sessions].sort((a, b) => 
        new Date(b.updatedAt).getTime() - new Date(a.updatedAt).getTime()
      )
    },
  },
  
  actions: {
    // 初始化用户信息
    initUserInfo(token: string, shopId: number, role: string, userName: string) {
      this.token = token
      this.shopId = shopId
      this.role = role
      this.userName = userName
    },
    
    // 发送消息
    async sendMessage(message: string, imageUrl?: string) {
      if (!this.currentSessionId) {
        await this.createSession()
      }
      
      this.isLoading = true
      
      // 添加用户消息
      this.messages.push({
        id: Date.now().toString(),
        role: 'user',
        content: message,
        images: imageUrl ? [imageUrl] : [],
        timestamp: Date.now(),
      })
      
      // 创建助手消息（用于流式更新）
      const assistantMessage = {
        id: (Date.now() + 1).toString(),
        role: 'assistant' as const,
        content: '',
        steps: [],
        timestamp: Date.now(),
      }
      this.messages.push(assistantMessage)
      
      // 连接 SSE
      const eventSource = chatStream(message, this.currentSessionId!, imageUrl)
      
      eventSource.onmessage = (event) => {
        const data = JSON.parse(event.data) as SSEData
        
        switch (data.type) {
          case 'thinking':
          case 'processing':
            assistantMessage.steps!.push(data)
            break
          case 'answer':
            assistantMessage.content += data.content
            break
          case 'done':
            this.isLoading = false
            eventSource.close()
            break
          case 'error':
            assistantMessage.content = `错误: ${data.content}`
            this.isLoading = false
            eventSource.close()
            break
        }
      }
      
      eventSource.onerror = () => {
        this.isLoading = false
        eventSource.close()
      }
    },
    
    // 创建会话
    async createSession() {
      const session = await createSession()
      this.sessions.unshift(session)
      this.currentSessionId = session.id
      this.messages = []
    },
    
    // 删除会话
    async deleteSession(sessionId: string) {
      await deleteSession(sessionId)
      this.sessions = this.sessions.filter(s => s.id !== sessionId)
      
      if (this.currentSessionId === sessionId) {
        this.currentSessionId = this.sessions[0]?.id || null
        if (this.currentSessionId) {
          await this.loadHistory(this.currentSessionId)
        } else {
          this.messages = []
        }
      }
    },
    
    // 加载会话历史
    async loadHistory(sessionId: string) {
      this.currentSessionId = sessionId
      this.messages = await getSessionHistory(sessionId)
    },
    
    // 切换全屏
    toggleFullscreen() {
      this.isFullscreen = !this.isFullscreen
    },
    
    // 切换会话列表
    toggleSessionList() {
      this.showSessionList = !this.showSessionList
    },
  },
})
```

## Docker 部署配置

### Dockerfile

```dockerfile
# 构建阶段
FROM node:18-alpine as build
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

# 运行阶段
FROM nginx:alpine
COPY --from=build /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

### docker-compose.yml

```yaml

services:
```

### nginx.conf

```nginx
server {
    listen 80;
    server_name localhost;
    
    root /usr/share/nginx/html;
    index index.html;
    
    # gzip 压缩
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;
    
    # API 代理
    location /api/ {
        proxy_pass http://ai-service:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        
        # SSE 支持
        proxy_http_version 1.1;
        proxy_set_header Connection '';
        proxy_buffering off;
        proxy_cache off;
        chunked_transfer_encoding off;
    }
    
    # Vue Router history 模式
    location / {
        try_files $uri $uri/ /index.html;
    }
    
    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}
```

## iframe 集成示例

### 后台系统引入

```html
<!-- 在后台系统中引入店铺助手 -->
<template>
  <div>
    <!-- 其他内容 -->
    
    <!-- 店铺助手悬浮按钮 -->
    <div class="chat-button" @click="showChat = true">
      <span>💬</span>
    </div>
    
    <!-- 店铺助手弹窗 -->
    <div v-if="showChat" class="chat-iframe-container">
      <iframe 
        ref="chatIframe"
        :src="chatUrl"
        @load="handleIframeLoad"
      />
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue'

const showChat = ref(false)
const chatIframe = ref(null)
const chatUrl = ref('http://localhost:3000')

// iframe 加载完成后发送用户信息
function handleIframeLoad() {
  const token = localStorage.getItem('token')
  const shopId = localStorage.getItem('shopId')
  const role = localStorage.getItem('role')
  const userName = localStorage.getItem('userName')
  
  chatIframe.value.contentWindow.postMessage({
    type: 'init',
    token,
    shopId: parseInt(shopId),
    role,
    userName
  }, '*')
}

// 监听店铺助手消息
function handleMessage(event) {
  if (event.data.type === 'close') {
    showChat.value = false
  }
}

onMounted(() => {
  window.addEventListener('message', handleMessage)
})

onUnmounted(() => {
  window.removeEventListener('message', handleMessage)
})
</script>
```
